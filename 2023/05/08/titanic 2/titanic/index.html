3. Data Fillup

# 存在空缺值的列：  'Age'  'Cabin'  'Embarked' 
# 'Cabin' missing ~77%, drop it
train_df.drop(columns='Cabin', inplace=True)
test_df.drop(columns='Cabin', inplace=True)
# 存在空缺值的列：  'Age'  'Cabin'  'Embarked' 
# 'Embarked' missing 2value,~0.22%
# 用上一条数据来填充，使用ffill填充'Embarked'
train_df['Embarked'] = train_df['Embarked'].fillna(method='ffill')
# 存在空缺值的列：  'Age'  'Cabin'  'Embarked' 
# 'Age' missing ~19.87% 
# 使用随机森林填充'Age'
from sklearn.ensemble import RandomForestRegressor
 
def set_missing_age(df):
    age_df = df[['Age','Pclass','SibSp','Parch']]
    
    #乘客分成已知年龄和未知年龄两个部分
    known_age = age_df[age_df['Age'].notnull()].values
    unknown_age = age_df[age_df['Age'].isnull()].values
    
    # 目标数据y,取第一列‘age'
    y = known_age[:,0]
    
    # 特征属性数据x，取非‘age'的列
    x = known_age[:,1:]
    
    # 利用随机森林进行拟合
    rfr = RandomForestRegressor(random_state=0,n_estimators=2000,n_jobs=-1)
    rfr.fit(x,y)
    
    # 利用训练的模型进行预测
    predictedAges = rfr.predict(unknown_age[:,1::])
#     print(predictedAges)
    
    # 填补缺失的原始数据
    df.loc[(df['Age'].isnull()),'Age'] = predictedAges
    return df
